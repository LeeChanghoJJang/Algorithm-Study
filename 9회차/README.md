pos = [list(map(int, input().split())) for _ in range(n)]

        dist = ((x1-x2)**2 + (y1-y2)**2)**0.5

    adj[i-1][j-1] = adj[j-1][i-1] = 0

MST = [0] * n
Q = []; heappush(Q, (0, 0)); ans = 0
while Q:
    weight, now = heappop(Q)
    if MST[now]: continue
    MST[now] = 1
    ans += weight

    for next in range(n):
        if now != next and not MST[next]:
            heappush(Q, (adj[now][next], next))
